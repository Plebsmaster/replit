const answers = [
  { text: 'Elegant 1', nextSlide: 'Slide4', dbValue: null },
  { text: 'Elegant 2', nextSlide: 'Slide5', dbValue: null },
  { text: 'Elegant 3', nextSlide: 'Slide7', dbValue: 'Elegant 3.' },
  { text: 'Elegant 4', nextSlide: 'Slide6', dbValue: 'Elegant 4.' },
  { text: 'Elegant 5', nextSlide: 'Slide6', dbValue: 'Elegant 5.' }
];
// Database column: 'Template'
// Only store dbValue when it's not null
const handleAnswer = (answer) => {
  if (answer.dbValue) {
    updateDatabase('Template', answer.dbValue);
  }
  navigateToSlide(answer.nextSlide);
};
