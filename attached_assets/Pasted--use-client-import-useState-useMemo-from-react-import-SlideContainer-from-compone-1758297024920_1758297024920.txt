"use client"

import { useState, useMemo } from "react"
import { SlideContainer } from "@/components/ui/slide-container"
import { getTypographyClasses } from "@/lib/typography"
import type { StepProps } from "@/lib/form/steps"

type Country = "Nederland" | "Duitsland" | "België"

const COUNTRIES: Array<{ name: Country; icon: string; description?: string }> = [
  { name: "Nederland", icon: "NL" },
  { name: "Duitsland", icon: "DE" },
  { name: "België", icon: "BE" },
]

type Props = StepProps & {}

export default function Slide19({ updateFormData, formData, onNext }: Props) {
  const initial = (formData?.marketCountry as Country | undefined) ?? "Nederland"
  const [selectedCountry, setSelectedCountry] = useState<Country>(initial)

  const countries = useMemo(() => COUNTRIES, [])

  const handleCountrySelect = (country: Country) => {
    setSelectedCountry(country)

    // Persist to global form data
    updateFormData({ marketCountry: country })

    // Keep your localStorage behavior
    try {
      localStorage.setItem("salonid:selectedCountry", country)
      localStorage.setItem("salonid:dateISO", new Date().toISOString())
    } catch {}

    // Auto-progress after selection (microtask = snappy + safe)
    queueMicrotask(() => onNext())
  }

  return (
    <SlideContainer width="wide">
      <section>
        <h2 className={getTypographyClasses("title", { alignment: "left" })}>
          In welk land worden de producten verkocht?
        </h2>

        <div className="max-w-[760px] space-y-4 mb-8">
          <p className={getTypographyClasses("paragraph", { alignment: "left" })}>
            Voor een optimale service en naleving van de regelgeving, vragen wij in welk land jouw producten
            voornamelijk worden verkocht. Dit helpt ons ervoor te zorgen dat jouw producten voldoen aan de
            specifieke wettelijke eisen en normen van dat land.
          </p>
        </div>

        <div className="mb-8">
          <h3 className="text-lg font-semibold text-gray-900 mb-4">Regels en Voorschriften</h3>
          <div className="text-gray-700 text-sm space-y-3">
            <p>
              Elke markt heeft zijn eigen set van regels en voorschriften met betrekking tot de etikettering,
              ingrediënten en verpakking van cosmetische producten. Door te weten waar jouw producten
              voornamelijk worden verkocht, kunnen wij:
            </p>
            <ul className="space-y-2 ml-4">
              <li>• Zorgen voor de juiste naleving van lokale wet- en regelgeving.</li>
              <li>• Jouw producten afstemmen op de voorkeuren en verwachtingen van consumenten in dat land.</li>
              <li>• De vereiste certificeringen en goedkeuringen verkrijgen.</li>
            </ul>
          </div>
        </div>

        <div className="mb-6">
          <h3 className="text-xl font-bold text-gray-900 text-left mb-4">Maak hier je keuze</h3>
        </div>

        <div className="space-y-3">
          {countries.map((country) => (
            <div
              key={country.name}
              className={`p-3 rounded-lg border cursor-pointer transition-all hover:shadow-sm hover:scale-105 ${
                selectedCountry === country.name ? "border-black bg-gray-50" : "border-gray-300 hover:border-gray-400"
              }`}
              onClick={() => handleCountrySelect(country.name)}
            >
              <div className="flex justify-between items-center">
                <div className="flex items-center space-x-3">
                  <span className="text-2xl font-bold bg-gray-200 rounded px-2 py-1 text-xs">{country.icon}</span>
                  <div>
                    <span className="text-gray-800 font-medium">{country.name}</span>
                  </div>
                </div>
                {selectedCountry === country.name && (
                  <div className="w-3 h-3 rounded-full bg-black flex-shrink-0" />
                )}
              </div>
            </div>
          ))}
        </div>
      </section>
    </SlideContainer>
  )
}
